CREATE OR REPLACE Procedure PF_InsertIncomeSource
    ( 
      
       incSource   nvarchar2  
           
       
      )
    IS
      SeqNumber integer;
    begin 
    
    
     select nvl(Count(IS_CODE),0) into SeqNumber from PF_INCOME_SOURCE;
            SeqNumber :=SeqNumber +1;
                       
        Insert into  PF_INCOME_SOURCE(IS_CODE,DESCRIPTION)
        values (SeqNumber,incSource);
    
       commit;
        
             
       
    end;
/
