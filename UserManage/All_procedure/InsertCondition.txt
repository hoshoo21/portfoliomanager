CREATE OR REPLACE Procedure PF_InsertCondtion
    ( 
      
       cond nvarchar2  ,
         act   nvarchar2  
       
      )
    IS
      SeqNumber integer;
    begin 
    
    
     select nvl(Count(CONDITION_CODE),0) into SeqNumber from PF_CONDITON;
            SeqNumber :=SeqNumber +1;
                       
        Insert into  PF_CONDITON(CONDITION_CODE,DESCRIPTION,ACTIVE)
        values (SeqNumber,cond,act);
    
       commit;
        
             
       
    end;
/
