CREATE OR REPLACE Procedure PF_InsertValue
    ( 
       val nvarchar2,  
      act nvarchar2  
           
       
      )
    IS
      SeqNumber integer;
    begin 
    
    
     select nvl(Count(VALUE_CODE),0) into SeqNumber from PF_VALUES;
            SeqNumber :=SeqNumber +1;
                       
        Insert into  PF_VALUES(VALUE_CODE,DESCRIPTION,ACTIVE)
        values (SeqNumber,val,act );
    
       commit;
        
             
       
    end;
/
